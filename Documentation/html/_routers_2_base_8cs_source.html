282  private static void CleanDeadRoutes(RoutingEvent EventType)
283  {
284  // Without the check this is supposedly at most a 3(O(n)) operation.
285  // With the check this ranges from a max of 4(O(n)) to 1 O(n) operation.
286  if (TablesExist && EventIsPopulated(EventType) && TableIsPolluted(EventType))
287  {
288  Route[] DeadRoutes = Array.FindAll(RouteTable[EventType].ToArray(), x => x.IsDead);
289  Array.ForEach(DeadRoutes, x => {DeregisterRoute(x); DetachAddress(x);});
290  TablesExist = (TablesExist? DeconstructTables() : TablesExist);
291  }
292  }
